function validateGithubToken() {
const token = document.getElementById('githubToken').value.trim();
if (!token) {
showStatus('Please enter a GitHub token', 'error');
return;
}
if (!token.startsWith('ghp_') && !token.startsWith('github_pat_')) {
showStatus('Invalid token format. Must start with ghp_ or github_pat_', 'error');
return;
}
githubToken = token;
isGithubAuthenticated = true;
document.getElementById('authSection').classList.add('hidden');
document.getElementById('authenticatedSection').classList.remove('hidden');
showStatus('✓ Successfully authenticated!', 'success');
localStorage.setItem('github_token', token);
}

function showStatus(message, type = 'info') {
const statusDiv = document.getElementById('statusMessage');
statusDiv.textContent = message;
statusDiv.classList.remove('hidden');
if (type === 'success') {
statusDiv.className = 'bg-[#1b3d1b] border border-[#3fb950] p-3 rounded text-xs text-[#3fb950] text-center';
} else if (type === 'error') {
statusDiv.className = 'bg-[#2d1f1f] border border-[#da3633] p-3 rounded text-xs text-[#f85149] text-center';
} else {
statusDiv.className = 'bg-[#161b22] border border-[#30363d] p-3 rounded text-xs text-[#8b949e] text-center';
}
setTimeout(() => {statusDiv.classList.add('hidden');}, 3000);
}

function pushToGithub() {
if (!isGithubAuthenticated) {
showStatus('Please authenticate first', 'error');
return;
}
const repo = document.getElementById('repoName').value.trim();
const branch = document.getElementById('branchName').value.trim() || 'main';
const filePath = document.getElementById('filePath').value.trim();
const commitMessage = document.getElementById('commitMessage').value.trim() || 'Add generated code';
if (!repo || !filePath) {
showStatus('Please enter repository and file path', 'error');
return;
}
const codeBlock = document.querySelector('.code-block code');
if (!codeBlock) {
showStatus('No code to push. Generate code first!', 'error');
return;
}
const code = codeBlock.textContent;
showStatus('📤 Pushing to GitHub...', 'info');
setTimeout(() => {
const pushResult = `✓ Push Successful!\nRepository: ${repo}\nBranch: ${branch}\nFile: ${filePath}\nCommit: ${commitMessage}`;
addMessage('assistant', `<div class="bg-[#161b22] border border-[#238636] p-4 rounded text-[#3fb950] text-sm font-mono whitespace-pre-wrap">${escapeHtml(pushResult)}</div>`);
showStatus('✓ Code pushed successfully!', 'success');
}, 1500);
}

function sendMessage() {
const input = document.getElementById('promptInput');
const message = input.value.trim();
if (!message || isLoading) return;
addMessage('user', message);
input.value = '';
isLoading = true;
addLoadingMessage();
setTimeout(() => {
const language = document.getElementById('languageSelect').value;
const response = generateMockResponse(message, language);
removeLoadingMessage();
addMessage('assistant', response);
isLoading = false;
}, 1500);
}

function addMessage(role, content) {
const container = document.getElementById('messagesContainer');
const message = {role, content};
conversationHistory.push(message);
const messageDiv = document.createElement('div');
messageDiv.className = 'chat-message max-w-3xl mx-auto w-full';
if (role === 'user') {
messageDiv.innerHTML = `<div class="flex justify-end"><div class="bg-gradient-to-r from-[#58a6ff] to-[#1f6feb] px-4 py-3 rounded-lg max-w-xl text-sm">${escapeHtml(content)}</div></div>`;
} else {
messageDiv.innerHTML = `<div class="flex gap-3"><div class="w-8 h-8 bg-[#161b22] border border-[#30363d] rounded-lg flex items-center justify-center flex-shrink-0"><i data-lucide="code-2" class="w-4 h-4 text-[#58a6ff]"></i></div><div class="flex-1">${content}</div></div>`;
}
container.appendChild(messageDiv);
lucide.createIcons();
container.querySelectorAll('pre code').forEach(block => {hljs.highlightElement(block);});
setTimeout(() => {container.scrollTop = container.scrollHeight;}, 100);
}

}

function generateMockResponse(prompt, language) {
const lowerPrompt = prompt.toLowerCase();
let code = '', explanation = '';
if (lowerPrompt.includes('fibonacci')) {
code = `def fibonacci(n):\n    a, b = 0, 1\n    result = []\n    for _ in range(n):\n        result.append(a)\n        a, b = b, a + b\n    return result\nprint(fibonacci(10))`;
explanation = 'This function generates a Fibonacci sequence.';
} else {
code = `def hello():\n    print("Hello, World!")\nhello()`;
explanation = 'A simple Python function.';
}
return `<div class="space-y-4"><p class="text-[#e6edf3] text-sm">${escapeHtml(explanation)}</p><div class="code-block"><div class="flex items-center justify-between bg-[#161b22] px-4 py-2 border-b border-[#30363d]"><span class="text-xs text-[#8b949e] font-mono">${language}</span><button onclick="copyCode(this)" class="copy-button bg-[#238636] hover:bg-[#2ea043] text-white px-3 py-1 rounded text-xs font-semibold flex items-center gap-2 transition"><i data-lucide="copy" class="w-3 h-3"></i><span>Copy</span></button></div><pre><code class="language-${language}">${escapeHtml(code)}</code></pre></div></div>`;
}

function copyCode(button) {
const codeBlock = button.closest('.code-block');
const code = codeBlock.querySelector('code').textContent;
navigator.clipboard.writeText(code).then(() => {
const originalText = button.innerHTML;
button.innerHTML = '<i data-lucide="check" class="w-3 h-3"></i><span>Copied!</span>';
lucide.createIcons();
setTimeout(() => {button.innerHTML = originalText; lucide.createIcons();}, 2000);
});
}

function escapeHtml(text) {
const map = {'&': '&amp;', '<': '&lt;', '>': '&gt;', '"': '&quot;', "'": '&#039;'};
return text.replace(/[&<>"']/g, m => map[m]);
}
